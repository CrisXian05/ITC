document.addEventListener("DOMContentLoaded", () => { const content = document.getElementById("content"); // Menu Item Handlers document.getElementById("nth-integer").addEventListener("click", () => { const n = prompt("Enter the position of the integer (n):"); if (n && !isNaN(n)) { content.innerHTML = `

The ${n}th integer is: ${parseInt(n)}

`; } else { content.innerHTML = "

Please enter a valid number!

"; } }); document.getElementById("prime-checker").addEventListener("click", () => { const num = prompt("Enter a number to check if it's prime:"); if (num && !isNaN(num)) { const isPrime = (n) => { if (n < 2) return false; for (let i = 2; i <= Math.sqrt(n); i++) { if (n % i === 0) return false; } return true; }; content.innerHTML = `

${num} is ${isPrime(num) ? "a prime number" : "not a prime number"}.

`; } else { content.innerHTML = "

Please enter a valid number!

"; } }); document.getElementById("factorial-calc").addEventListener("click", () => { const num = prompt("Enter a number to calculate its factorial:"); if (num && !isNaN(num)) { const factorial = (n) => (n <= 1 ? 1 : n * factorial(n - 1)); content.innerHTML = `

The factorial of ${num} is: ${factorial(num)}

`; } else { content.innerHTML = "

Please enter a valid number!

"; } }); });
